334  void allocate_tiles() {
335  tiles.clear();
336 
337  uindex_t n_tile_columns = grid_range[0] / tile_width;
338  if (grid_range[0] % tile_width != 0) {
339  n_tile_columns++;
340  }
341  uindex_t n_tile_rows = grid_range[1] / tile_height;
342  if (grid_range[1] % tile_height != 0) {
343  n_tile_rows++;
344  }
345 
346  tiles.reserve(n_tile_columns + 2);
347  for (uindex_t i_column = 0; i_column < n_tile_columns + 2; i_column++) {
348  std::vector<Tile> column;
349  column.reserve(n_tile_rows + 2);
350  for (uindex_t i_row = 0; i_row < n_tile_rows + 2; i_row++) {
351  column.push_back(Tile());
352  }
353  tiles.push_back(column);
354  }
355  }
