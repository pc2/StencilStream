253  void copy_part(
254  cl::sycl::accessor<T, 2, cl::sycl::access::mode::read_write, cl::sycl::access::target::host_buffer> accessor,
255  Part part,
256  cl::sycl::id<2> global_offset,
257  bool buffer_to_part)
258  {
259  cl::sycl::id<2> offset = global_offset + get_part_offset(part);
260  if (offset[0] >= accessor.get_range()[0] || offset[1] >= accessor.get_range()[1])
261  {
262  // Nothing to do here. There is no data in the buffer for this part.
263  return;
264  }
265 
266  auto part_ac = (*this)[part].template get_access<cl::sycl::access::mode::read_write>();
267  auto part_range = get_part_range(part);
268 
269  uindex_t i_burst = 0;
270  uindex_t i_cell = 0;
271  for (uindex_t c = 0; c < part_range[0]; c++)
272  {
273  for (uindex_t r = 0; r < part_range[1]; r++)
274  {
275  if (c + offset[0] < accessor.get_range()[0] && r + offset[1] < accessor.get_range()[1])
276  {
277  if (buffer_to_part)
278  {
279  part_ac[i_burst][i_cell] = accessor[c + offset[0]][r + offset[1]];
280  }
281  else
282  {
283  accessor[c + offset[0]][r + offset[1]] = part_ac[i_burst][i_cell];
284  }
285  }
286 
287  if (i_cell == burst_length - 1)
288  {
289  i_cell = 0;
290  i_burst++;
291  }
292  else
293  {
294  i_cell++;
295  }
296  }
297  }
298  }
