264  void copy_part(cl::sycl::accessor<T, 2, cl::sycl::access::mode::read_write,
265  cl::sycl::access::target::host_buffer>
266  accessor,
267  Part part, cl::sycl::id<2> global_offset, bool buffer_to_part) {
268  cl::sycl::id<2> offset = global_offset + get_part_offset(part);
269  if (offset[0] >= accessor.get_range()[0] || offset[1] >= accessor.get_range()[1]) {
270  // Nothing to do here. There is no data in the buffer for this part.
271  return;
272  }
273 
274  auto part_ac = (*this)[part].template get_access<cl::sycl::access::mode::read_write>();
275  auto part_range = get_part_range(part);
276 
277  uindex_t i_burst = 0;
278  uindex_t i_cell = 0;
279  for (uindex_t c = 0; c < part_range[0]; c++) {
280  for (uindex_t r = 0; r < part_range[1]; r++) {
281  if (c + offset[0] < accessor.get_range()[0] &&
282  r + offset[1] < accessor.get_range()[1]) {
283  if (buffer_to_part) {
284  part_ac[i_burst][i_cell] = accessor[c + offset[0]][r + offset[1]];
285  } else {
286  accessor[c + offset[0]][r + offset[1]] = part_ac[i_burst][i_cell];
287  }
288  }
289 
290  if (i_cell == burst_length - 1) {
291  i_cell = 0;
292  i_burst++;
293  } else {
294  i_cell++;
295  }
296  }
297  }
298  }
