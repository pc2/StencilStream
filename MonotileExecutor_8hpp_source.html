77  void set_input(cl::sycl::buffer<T, 2> input_buffer) override {
78  if (input_buffer.get_range()[0] > tile_width && input_buffer.get_range()[1] > tile_height) {
79  throw std::range_error("The grid is bigger than the tile. The monotile architecture "
80  "requires that grid ranges are smaller or equal to the tile "
81  "range");
82  }

84  tile_buffer = cl::sycl::buffer<T, 2>(input_buffer.get_range());

86  for (uindex_t c = 0; c < input_buffer.get_range()[0]; c++) {
87  for (uindex_t r = 0; r < input_buffer.get_range()[1]; r++) {
88  tile_ac[c][r] = in_ac[c][r];

100  for (uindex_t r = 0; r < tile_buffer.get_range()[1]; r++) {

110  void run(uindex_t n_generations) override {
111  using in_pipe = cl::sycl::pipe<class monotile_in_pipe, T>;
112  using out_pipe = cl::sycl::pipe<class monotile_out_pipe, T>;
113  using ExecutionKernelImpl =
114  monotile::ExecutionKernel<TransFunc, T, stencil_radius, pipeline_length, tile_width,
115  tile_height, in_pipe, out_pipe>;
116 
117  cl::sycl::queue &queue = this->get_queue();
118 
119  uindex_t target_i_generation = this->get_i_generation() + n_generations;
120  uindex_t grid_width = tile_buffer.get_range()[0];
121  uindex_t grid_height = tile_buffer.get_range()[1];
122 
123  while (this->get_i_generation() < target_i_generation) {
124  cl::sycl::buffer<T, 2> out_buffer(tile_buffer.get_range());
125 
126  queue.submit([&](cl::sycl::handler &cgh) {
127  auto ac = tile_buffer.template get_access<cl::sycl::access::mode::read>(cgh);
128  T halo_value = this->get_halo_value();
129 
130  cgh.single_task<class MonotileInputKernel>([=]() {
131  [[intel::loop_coalesce(2)]] for (uindex_t c = 0; c < tile_width; c++) {
132  for (uindex_t r = 0; r < tile_height; r++) {
133  T value;
134  if (c < grid_width && r < grid_height) {
135  value = ac[c][r];
136  } else {
137  value = halo_value;
138  }
139 
140  in_pipe::write(value);
141  }
142  }
143  });
144  });
145 
146  cl::sycl::event computation_event = queue.submit([&](cl::sycl::handler &cgh) {
147  cgh.single_task(ExecutionKernelImpl(
148  this->get_trans_func(), this->get_i_generation(), target_i_generation,
149  grid_width, grid_height, this->get_halo_value()));
150  });
151 
152  queue.submit([&](cl::sycl::handler &cgh) {
153  auto ac =
154  out_buffer.template get_access<cl::sycl::access::mode::discard_write>(cgh);
155  T halo_value = this->get_halo_value();
156 
157  cgh.single_task<class MonotileOutputKernel>([=]() {
158  [[intel::loop_coalesce(2)]] for (uindex_t c = 0; c < tile_width; c++) {
159  for (uindex_t r = 0; r < tile_height; r++) {
160  T value = out_pipe::read();
161  if (c < grid_width && r < grid_height) {
162  ac[c][r] = value;
163  }
164  }
165  }
166  });
167  });
168 
169  tile_buffer = out_buffer;
170 
171  if (this->is_runtime_analysis_enabled()) {
172  this->get_runtime_sample().add_pass(computation_event);
173  }
174 
175  this->inc_i_generation(
176  std::min(target_i_generation - this->get_i_generation(), pipeline_length));
177  }
178  }
