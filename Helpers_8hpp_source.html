17 inline cl::sycl::range<2> burst_partitioned_range(uindex_t width, uindex_t height, uindex_t burst_length)
18 {
19  uindex_t nCells = width * height;
20  cl::sycl::range<2> range(nCells / burst_length, burst_length);
21  if (nCells % burst_length != 0)
22  {
23  range[0] += 1;
24  }
25  return range;
26 }
