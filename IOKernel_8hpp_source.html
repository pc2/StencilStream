135  template <typename Action> void run(Action action) {
136  static_assert(std::is_invocable<Action, Accessor &, uindex_t, uindex_t>::value);
137 
138  uindex_t burst_i[n_buffers] = {0};
139  uindex_t cell_i[n_buffers] = {0};
140 
141  for (uindex_t c = 0; c < n_columns; c++) {
142  uindex_t buffer_i = 0;
143  uindex_t next_bound = get_buffer_height(0);
144  for (uindex_t r = 0; r < n_rows; r++) {
145  if (r == next_bound) {
146  buffer_i++;
147  next_bound += get_buffer_height(buffer_i);
148  }
149 
150  action(accessor[buffer_i], burst_i[buffer_i], cell_i[buffer_i]);
151  if (cell_i[buffer_i] == burst_length - 1) {
152  cell_i[buffer_i] = 0;
153  burst_i[buffer_i]++;
154  } else {
155  cell_i[buffer_i]++;
156  }
157  }
158  }
159  }
