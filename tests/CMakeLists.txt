add_compile_definitions(STENCIL_INDEX_WIDTH=32)

    monotile/StencilUpdate.cpp
    tiling/Grid.cpp
    tiling/StencilUpdate.cpp
    constraint_test.cpp
)

add_executable(unit_test ${UNIT_TEST_SOURCES})
target_link_libraries(unit_test PRIVATE Catch2::Catch2WithMain)
