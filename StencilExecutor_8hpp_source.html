87  void run(uindex_t n_generations) override {
88  using in_pipe = cl::sycl::pipe<class tiling_in_pipe, T>;
89  using out_pipe = cl::sycl::pipe<class tiling_out_pipe, T>;
90  using ExecutionKernelImpl =
91  tiling::ExecutionKernel<TransFunc, T, stencil_radius, pipeline_length, tile_width,
92  tile_height, in_pipe, out_pipe>;
93 
94  cl::sycl::queue &queue = this->get_queue();
95 
96  uindex_t target_i_generation = this->get_i_generation() + n_generations;
97  uindex_t grid_width = input_grid.get_grid_range().c;
98  uindex_t grid_height = input_grid.get_grid_range().r;
99 
100  while (this->get_i_generation() < target_i_generation) {
101  GridImpl output_grid = input_grid.make_output_grid();
102 
103  std::vector<cl::sycl::event> events;
104  events.reserve(input_grid.get_tile_range().c * input_grid.get_tile_range().r);
105 
106  for (uindex_t c = 0; c < input_grid.get_tile_range().c; c++) {
107  for (uindex_t r = 0; r < input_grid.get_tile_range().r; r++) {
108  input_grid.template submit_tile_input<in_pipe>(queue, UID(c, r));
109 
110  cl::sycl::event computation_event = queue.submit([&](cl::sycl::handler &cgh) {
111  cgh.single_task(ExecutionKernelImpl(
112  this->get_trans_func(), this->get_i_generation(), target_i_generation,
113  c * tile_width, r * tile_height, grid_width, grid_height,
114  this->get_halo_value()));
115  });
116  events.push_back(computation_event);
117 
118  output_grid.template submit_tile_output<out_pipe>(queue, UID(c, r));
119  }
120  }
121 
122  input_grid = output_grid;
123 
124  if (this->is_runtime_analysis_enabled()) {
125  double earliest_start = std::numeric_limits<double>::max();
126  double latest_end = std::numeric_limits<double>::min();
127 
128  for (cl::sycl::event event : events) {
129  earliest_start = std::min(earliest_start, RuntimeSample::start_of_event(event));
130  latest_end = std::max(latest_end, RuntimeSample::end_of_event(event));
131  }
132  this->get_runtime_sample().add_pass(latest_end - earliest_start);
133  }
134 
135  this->inc_i_generation(
136  std::min(target_i_generation - this->get_i_generation(), pipeline_length));
137  }
138  }
139 
140  private:
141  using GridImpl = tiling::Grid<T, tile_width, tile_height, halo_radius, burst_length>;
142  GridImpl input_grid;
143 };
