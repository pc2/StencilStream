85  void run(uindex_t n_generations)
86  {
87  using in_pipe = cl::sycl::pipe<class in_pipe_id, T>;
88  using out_pipe = cl::sycl::pipe<class out_pipe_id, T>;
89  using ExecutionKernelImpl = ExecutionKernel<TransFunc, T, stencil_radius, pipeline_length, tile_width, tile_height, in_pipe, out_pipe>;
90 
91  if (!this->queue.has_value())
92  {
93  select_emulator(false);
94  }
95 
96  cl::sycl::queue queue = *(this->queue);
97 
98  uindex_t n_passes = n_generations / pipeline_length;
99  if (n_generations % pipeline_length != 0)
100  {
101  n_passes += 1;
102  }
103 
104  if (runtime_analysis_enabled)
105  {
106  runtime_sample = std::nullopt;
107  runtime_sample = RuntimeSample(n_passes, input_grid.get_tile_range().c, input_grid.get_tile_range().r);
108  }
109 
110  uindex_t grid_width = input_grid.get_grid_range().c;
111  uindex_t grid_height = input_grid.get_grid_range().r;
112 
113  for (uindex_t i = 0; i < n_passes; i++)
114  {
115  Grid output_grid = input_grid.make_output_grid();
116  uindex_t n_generations_per_pass = pipeline_length;
117  if (i == n_passes - 1 && n_generations % pipeline_length != 0)
118  {
119  n_generations_per_pass = n_generations % pipeline_length;
120  }
121 
122  for (uindex_t c = 0; c < input_grid.get_tile_range().c; c++)
123  {
124  for (uindex_t r = 0; r < input_grid.get_tile_range().r; r++)
125  {
126  input_grid.template submit_tile_input<in_pipe>(queue, UID(c, r));
127 
128  cl::sycl::event computation_event = queue.submit([&](cl::sycl::handler &cgh)
129  { cgh.single_task(ExecutionKernelImpl(
130  trans_func,
131  i_generation,
132  n_generations_per_pass,
133  c * tile_width,
134  r * tile_height,
135  grid_width,
136  grid_height,
137  halo_value)); });
138 
139  if (runtime_analysis_enabled)
140  {
141  runtime_sample->add_event(computation_event, i, c, r);
142  }
143 
144  output_grid.template submit_tile_output<out_pipe>(queue, UID(c, r));
145  }
146  }
147  input_grid = output_grid;
148  i_generation += n_generations_per_pass;
149  }
150  }
