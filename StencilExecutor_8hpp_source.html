35  void run(uindex_t n_generations)
36  {
37  using in_pipe = cl::sycl::pipe<class in_pipe_id, T>;
38  using out_pipe = cl::sycl::pipe<class out_pipe_id, T>;
39  using ExecutionKernelImpl = ExecutionKernel<TransFunc, T, stencil_radius, pipeline_length, tile_width, tile_height, in_pipe, out_pipe>;
40 
41  if (!this->queue.has_value())
42  {
43  select_emulator(false);
44  }
45 
46  cl::sycl::queue queue = *(this->queue);
47 
48  uindex_t n_passes = n_generations / pipeline_length;
49  if (n_generations % pipeline_length != 0)
50  {
51  n_passes += 1;
52  }
53 
54  if (runtime_analysis_enabled)
55  {
56  runtime_sample = RuntimeSample(n_passes, input_grid.get_tile_range().c, input_grid.get_tile_range().r);
57  }
58 
59  for (uindex_t i = 0; i < n_passes; i++)
60  {
61  Grid output_grid = input_grid.make_output_grid();
62  uindex_t n_generations_per_pass = pipeline_length;
63  if (i == n_passes - 1 && n_generations % pipeline_length != 0)
64  {
65  n_generations_per_pass = n_generations % pipeline_length;
66  }
67 
68  for (uindex_t c = 0; c < input_grid.get_tile_range().c; c++)
69  {
70  for (uindex_t r = 0; r < input_grid.get_tile_range().r; r++)
71  {
72  input_grid.template submit_tile_input<in_pipe>(queue, UID(c, r));
73 
74  cl::sycl::event computation_event = queue.submit([&](cl::sycl::handler &cgh) {
75  cgh.single_task(ExecutionKernelImpl(trans_func, i_generation, n_generations_per_pass, c * tile_width, r * tile_height));
76  });
77 
78  if (runtime_analysis_enabled)
79  {
80  runtime_sample->add_event(computation_event, i, c, r);
81  }
82 
83  output_grid.template submit_tile_output<out_pipe>(queue, UID(c, r));
84  }
85  }
86  input_grid = output_grid;
87  i_generation += n_generations_per_pass;
88  }
89  }
